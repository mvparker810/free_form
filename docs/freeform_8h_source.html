562/* ===== Generic gen+idx table: definitions ===== */
563#define FF__CLAMP_CAP_16(x) ((uint16_t)((x) > 0xFFFF ? 0xFFFF : (x)))
564
565#define FF_DEFINE_GENTABLE(PREFIX, PAYLOAD_T) \
566 static void PREFIX##TBL__grow(PREFIX##__table* t, uint16_t add) { \
567 if (add == 0) return; \
568 uint32_t new_cap32 = (uint32_t)t->cap + (uint32_t)add; \
569 uint16_t new_cap = FF__CLAMP_CAP_16(new_cap32); \
570 if (new_cap <= t->cap) return; /* hit 65535 */ \
571 PREFIX##__slot* new_slots = (PREFIX##__slot*)realloc( \
572 t->slots, sizeof(PREFIX##__slot) * new_cap); \
573 if (!new_slots) return; /* OOM: leave table unchanged */ \
574 /* initialize newly added range and link them into free list */ \
575 for (uint16_t i = t->cap; i < new_cap; ++i) { \
576 new_slots[i].gen = 1; \
577 new_slots[i].alive = false; \
578 new_slots[i].next_free = (uint16_t)(i + 1); \
579 new_slots[i].payload = PAYLOAD_T##_DEFAULT(); \
580 } \
581 new_slots[new_cap - 1].next_free = t->free_head; \
582 t->free_head = t->cap; \
583 t->slots = new_slots; \
584 t->cap = new_cap; \
585 } \
586 \
587 void PREFIX##TBL_init(PREFIX##__table* t, uint16_t initial_cap) { \
588 t->slots = NULL; \
589 t->cap = 0; \
590 t->alive_count = 0; \
591 t->free_head = FF_INVALID_INDEX; \
592 if (initial_cap) PREFIX##TBL__grow(t, initial_cap); \
593 } \
594 \
595 void PREFIX##TBL_free(PREFIX##__table* t) { \
596 free(t->slots); \
597 t->slots = NULL; \
598 t->cap = t->alive_count = 0; \
599 t->free_head = FF_INVALID_INDEX; \
600 } \
601 \
602 ff_GeneralHandle PREFIX##TBL_create(PREFIX##__table* t, const PAYLOAD_T* init) { \
603 if (t->free_head == FF_INVALID_INDEX) { \
604 /* grow (geometric) */ \
605 uint16_t add = (t->cap < 64) ? 64 : (t->cap >> 1); \
606 if (add == 0) add = 64; \
607 PREFIX##TBL__grow(t, add); \
608 if (t->free_head == FF_INVALID_INDEX) { \
609 return (ff_GeneralHandle){ FF_INVALID_INDEX, 0 }; \
610 } \
611 } \
612 uint16_t idx = t->free_head; \
613 PREFIX##__slot* s = &t->slots[idx]; \
614 t->free_head = s->next_free; \
615 s->alive = true; \
616 if (init) memcpy(&s->payload, init, sizeof(PAYLOAD_T)); \
617 else memset(&s->payload, 0, sizeof(PAYLOAD_T)); \
618 t->alive_count++; \
619 return (ff_GeneralHandle){ idx, s->gen }; \
620 } \
621 \
622 static inline bool PREFIX##TBL__valid(const PREFIX##__table* t, ff_GeneralHandle h) { \
623 return (h.idx != FF_INVALID_INDEX) && (h.idx < t->cap); \
624 } \
625 \
626 bool PREFIX##TBL_alive(const PREFIX##__table* t, ff_GeneralHandle h) { \
627 if (!PREFIX##TBL__valid(t, h)) return false; \
628 const PREFIX##__slot* s = &t->slots[h.idx]; \
629 return s->alive && (s->gen == h.gen); \
630 } \
631 \
632 PAYLOAD_T* PREFIX##TBL_get(PREFIX##__table* t, ff_GeneralHandle h) { \
633 if (!PREFIX##TBL_alive(t, h)) return NULL; \
634 return &t->slots[h.idx].payload; \
635 } \
636 \
637 const PAYLOAD_T* PREFIX##TBL_get_const(const PREFIX##__table* t, ff_GeneralHandle h) { \
638 if (!PREFIX##TBL_alive(t, h)) return NULL; \
639 return &t->slots[h.idx].payload; \
640 } \
641 \
642 bool PREFIX##TBL_destroy(PREFIX##__table* t, ff_GeneralHandle h) { \
643 if (!PREFIX##TBL_alive(t, h)) return false; \
644 PREFIX##__slot* s = &t->slots[h.idx]; \
645 s->alive = false; \
646 s->gen += 1u; /* bump generation to invalidate stale handles */ \
647 s->next_free = t->free_head; \
648 t->free_head = h.idx; \
649 if (t->alive_count) t->alive_count--; \
650 return true; \
651 } \
652 \
653 const ff_GeneralHandle PREFIX##_INVALIDHANDLE = { FF_INVALID_INDEX, 0u }; \

889// Evaluate the expression
890ff_float expr_evaluate(ff_Expr* expr, const ff_param__table *t) {
891
892
893
894 switch (expr->op_type) {
895 case OperatorType_CONST: return expr->value;
896 case OperatorType_PARAM: {
897 const ff_Parameter * p = ff_paramTBL_get_const(t, expr->param_H);
898 return p->def.v;
899 }
900 case OperatorType_EXTR_PARAM: return expr_evaluate(expr->a,t);
901 case OperatorType_ADD: return expr_evaluate(expr->a,t) + expr_evaluate(expr->b,t);
902 case OperatorType_SUB: return expr_evaluate(expr->a,t) - expr_evaluate(expr->b,t);
903 case OperatorType_MUL: return expr_evaluate(expr->a,t) * expr_evaluate(expr->b,t);
904 case OperatorType_DIV: return expr_evaluate(expr->a,t) / expr_evaluate(expr->b,t);
905 case OperatorType_SIN: return sin(expr_evaluate(expr->a,t));
906 case OperatorType_COS: return cos(expr_evaluate(expr->a,t));
907 case OperatorType_ASIN: return asin(expr_evaluate(expr->a,t));
908 case OperatorType_ACOS: return acos(expr_evaluate(expr->a,t));
909 case OperatorType_SQRT: return sqrt(expr_evaluate(expr->a,t));
910 case OperatorType_SQR: { ff_float val = expr_evaluate(expr->a,t); return val * val; }
911 }
912 return 0.0;
913}

917// Differentiate the expression with respect to a parameter
918ff_Expr* expr_derivative(ff_Expr* expr, ff_ParamHandle indp_param, bool protect_params) {
919 switch (expr->op_type) {
920 case OperatorType_CONST:
921 return exprInit_const(0.0);
922 case OperatorType_PARAM:
923 return ffParam_Equals(expr->param_H, indp_param) ?
924 exprInit_const(1.0) : exprInit_const(0.0);
925 case OperatorType_EXTR_PARAM:
926 return expr_derivative(expr->a, indp_param, protect_params);
927 case OperatorType_ADD:
928 return exprInit_op(OperatorType_ADD, expr_derivative(expr->a, indp_param, protect_params), expr_derivative(expr->b, indp_param, protect_params));
929 case OperatorType_SUB:
930 return exprInit_op(OperatorType_SUB, expr_derivative(expr->a, indp_param, protect_params), expr_derivative(expr->b, indp_param, protect_params));
931 case OperatorType_MUL:
932 return exprInit_op(OperatorType_ADD,
933 exprInit_op(OperatorType_MUL, expr_derivative(expr->a, indp_param, protect_params), TRY_EXTR_PARAM(expr->b)),
934 exprInit_op(OperatorType_MUL, TRY_EXTR_PARAM(expr->a), expr_derivative(expr->b, indp_param, protect_params))
935 );
936 case OperatorType_DIV:
937 return exprInit_op(OperatorType_DIV,
938 exprInit_op(OperatorType_SUB,
939 exprInit_op(OperatorType_MUL, expr_derivative(expr->a, indp_param, protect_params), TRY_EXTR_PARAM(expr->b)),
940 exprInit_op(OperatorType_MUL, TRY_EXTR_PARAM(expr->a), expr_derivative(expr->b, indp_param, protect_params))
941 ),
942 exprInit_op(OperatorType_MUL, TRY_EXTR_PARAM(expr->b), TRY_EXTR_PARAM(expr->b))
943 );
944 case OperatorType_SIN:
945 return exprInit_op(OperatorType_MUL, expr_derivative(expr->a, indp_param, protect_params), exprInit_op(OperatorType_COS, expr->a, NULL));
946 case OperatorType_COS:
947 return exprInit_op(OperatorType_MUL,
948 exprInit_op(OperatorType_MUL, exprInit_const(-1.0),
949 exprInit_op(OperatorType_SIN, expr->a, NULL)),
950 expr_derivative(expr->a, indp_param, protect_params));
951 case OperatorType_ASIN:

1158bool ffSketch_Solve(ff_Sketch* skt, double tolerance, uint32_t max_steps) {
1159
1160 ffSketch_tryRelink(skt);
1161
1162 uint16_t rows = skt->constraints.alive_count;
1163 uint16_t cols = skt->params.alive_count;
1164
1165 if (!(rows && cols)) return true;
1166
1167 double temp = 0.0;
1168 const double epsilon = 1e-10;
1169
1170 bool converged = false;
1171
1172
1173
1174
1175 for (uint32_t step = 0; step < max_steps; step++) {
1176
1177 printf("* Iteration (%d/%d)\n", step + 1, max_steps);
1178
1179
1180 //Calculate error of system. If we are converged we are done.
1181 if (ffSketch_calcError(skt, tolerance)) {
1182 converged = true;
1183 break;
1184 }
1185
1186 for (uint16_t i = 0; i < skt->constraints.alive_count; i++) {
1187 ff_Constraint* cons = skt->tmp_contraints[i];
1188 for (uint16_t p = 0; p < skt->params.alive_count; p++) {
1189 cons->JMR.dervs_y[p] = expr_evaluate(cons->JMR.dervs[p], &skt->params);
1190 printf("D= %f\n", cons->JMR.dervs_y[p]);
1191 }
1192 }
1193
1194 //Solve by least squares:
1195
1196 //Start
1197 for (int r = 0; r < rows; r++) {
1198 for (int c = 0; c < rows; c++) {
1199 double sum = 0.0;
1200 for (int i = 0; i < cols; i++) {
1201 double cV = skt->tmp_contraints[c]->JMR.dervs_y[i];
1202 double rV = skt->tmp_contraints[r]->JMR.dervs_y[i];
1203 if (cV == 0.0 || rV == 0.0) continue;
1204 sum += rV * cV;
1205 }
1206 skt->normal_mtr[r + c * rows] = sum;
1207 }
1208 }
1209
1210
1211 //Gaussian Solve
1212 for (int row = 0; row < rows; row++) {
1213 int pivot_row = row;
1214 double max_value = 0.0;
1215
1216 // Find the pivot row (max absolute value in column)
1217 for (int candidate_row = row; candidate_row < rows; candidate_row++) {
1218 if (candidate_row >= rows) ff_ERROR("Bounds error"); //todo will these bounds errors ever fire?
1219 if (fabs(skt->normal_mtr[candidate_row + row * rows]) > max_value) {
1220 max_value = fabs(skt->normal_mtr[candidate_row + row * rows]);
1221 pivot_row = candidate_row;
1222 }
1223 }
1224
1225 if (max_value < epsilon) {
1226 printf("Small pivot element: %f at row %d\n", max_value, row);
1227 continue;
1228 }
1229
1230 // Swap rows in the normal matrix
1231 for (int col = 0; col < rows; col++) {
1232 temp = skt->normal_mtr[row + col * rows];
1233 skt->normal_mtr[row + col * rows] = skt->normal_mtr[pivot_row + col * rows];
1234 skt->normal_mtr[pivot_row + col * rows] = temp;
1235 }
1236
1237 // Swap elements in the constraint error vector
1238 temp = skt->tmp_contraints[row]->JMR.err;
1239 skt->tmp_contraints[row ]->JMR.err = skt->tmp_contraints[pivot_row]->JMR.err;
1240 skt->tmp_contraints[pivot_row]->JMR.err = temp;
1241
1242 // Eliminate entries below the pivot
1243 for (int target_row = row + 1; target_row < rows; target_row++) {
1244 if(target_row >= rows) ff_ERROR("Bounds error"); //todo will these bounds errors ever fire?
1245 if (fabs(skt->normal_mtr[row + row * rows]) < epsilon) {
1246 printf("Division by zero at row %d\n", row);
1247 continue;
1248 }
1249 double coefficient = skt->normal_mtr[target_row + row * rows] / skt->normal_mtr[row + row * rows];
1250 for (int col = 0; col < rows; col++) {
1251 skt->normal_mtr[target_row + col* rows] -= skt->normal_mtr[row + col* rows] * coefficient;
1252 }
1253 skt->tmp_contraints[target_row]->JMR.err -= skt->tmp_contraints[row]->JMR.err * coefficient;
1254 }
1255 }
1256
1257
1258 //Back sub
1259 for (int row = rows - 1; row >= 0; row--) {
1260 if (fabs(skt->normal_mtr[row + row * rows]) < epsilon) {
1261 printf("Back substitution failed at row %d\n", row);
1262 continue;
1263 }
1264
1265 ff_float solution_value = skt->tmp_contraints[row]->JMR.err / skt->normal_mtr[row + row * rows];
1266 for (int prev_row = rows - 1; prev_row > row; prev_row--) {
1267 solution_value -= skt->itrm_sol[prev_row] * skt->normal_mtr[row + prev_row * rows] / skt->normal_mtr[row + row * rows];
1268 }
1269 skt->itrm_sol[row] = solution_value;
1270 }
1271
1272 //Update parameters based on this steps corrections
1273 for (int c = 0; c < cols; c++) {
1274 ff_float corr = 0.0;
1275 for (int r = 0; r < rows; r++) {
1276 corr += skt->itrm_sol[r] * skt->tmp_contraints[r]->JMR.dervs_y[c];
1277 }
1278 printf("Correction is %f\n", corr);
1279 skt->tmp_params[c]->def.v -= corr;
1280 }
1281
1282 printf("--==--==--==--\n\n\n");
1283
1284 } //For step in maxsteps
1285
1286
1287
1288
1289 //End of solving process.
1290 //TODO@ revert params here when failed
1291 return converged;
1292}
